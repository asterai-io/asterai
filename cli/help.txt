asterai CLI

commands:

- auth <api_key>
- env
  - init <name>
      create a new environment with <name>
  - publish
      publishes an environment to the asterai cloud
  - ls
      list all environments
  - add --env <environment-name> --plugin <plugin-name>
      add a plugin to the environment
  - remove --env <environment-name> --plugin <plugin-name>
      remove a plugin from the environment
  - set-var --env <environment-name> --var <env-var>
      set an environment variable, e.g. NAME=VALUE.
      to clear a var, leave the value empty i.e. NAME=
  - run <name>
      run a new environment instance
  - end <instance-id>
      end an environment instance
- plugin
  - init <name>: init a new local plugin project
  - publish: pushes the plugin to the registry
  - list: list all environments
